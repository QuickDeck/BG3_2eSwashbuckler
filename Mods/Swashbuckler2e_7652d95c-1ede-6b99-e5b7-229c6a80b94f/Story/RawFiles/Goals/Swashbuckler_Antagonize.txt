UsingSpellOnTarget((GUIDSTRING)_Character, (GUIDSTRING)_Target, (STRING)_Spell, _, _, (INTEGER)_StoryID)

AND

THEN

DB_AntagonizeCast(_Character, _Target, _StoryID)
THEN

IF
StatusRemoved((GUIDSTRING)_Character, "FRIGHTENED", (GUIDSTRING)_Causee, _)
AND
HasActiveStatus(_Character, "ANTAGONIZE", 1)
THEN
